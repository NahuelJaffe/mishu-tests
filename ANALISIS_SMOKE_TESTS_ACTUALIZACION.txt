📊 WORKFLOW RECOMENDADO PARA SMOKE TESTS
========================================

🎯 CREAR NUEVO WORKFLOW: smoke-tests-complete.yml
================================================

```yaml
name: Smoke Tests Complete - Excel STD_KPIs

on:
  push:
    branches: [ main, smoke-tests-complete ]
  pull_request:
    branches: [ main, smoke-tests-complete ]
  workflow_dispatch:
    inputs:
      test_scope:
        description: 'Scope of smoke tests to run'
        required: true
        default: 'core'
        type: choice
        options:
          - core
          - extended
          - all

jobs:
  smoke-tests-chromium:
    timeout-minutes: 30
    runs-on: ubuntu-latest
    
    steps:
    - uses: actions/checkout@v4
    
    - uses: actions/setup-node@v4
      with:
        node-version: '18'
        cache: 'npm'
    
    - name: Install dependencies
      run: npm ci
    
    - name: Install Playwright Browsers
      run: npx playwright install --with-deps
    
    - name: Run Smoke Tests Complete - Chromium
      run: |
        echo "🔥 Ejecutando Smoke Tests Complete en Chromium"
        
        SCOPE="${{ github.event.inputs.test_scope || 'core' }}"
        
        case $SCOPE in
          "core")
            echo "🧪 Ejecutando TC-001 a TC-011 (Core smoke tests)"
            npx playwright test tests/smoke-tests-complete.spec.js --grep "TC-00[1-9]|TC-01[0-1]" --project=chromium --reporter=html --timeout=120000 --max-failures=5 --retries=1
            ;;
          "extended")
            echo "🧪 Ejecutando TC-001 a TC-014 (Extended smoke tests)"
            npx playwright test tests/smoke-tests-complete.spec.js --grep "TC-00[1-9]|TC-01[0-4]" --project=chromium --reporter=html --timeout=120000 --max-failures=5 --retries=1
            ;;
          "all")
            echo "🧪 Ejecutando TC-001 a TC-016 (All smoke tests)"
            npx playwright test tests/smoke-tests-complete.spec.js --project=chromium --reporter=html --timeout=120000 --max-failures=5 --retries=1
            ;;
        esac
      env:
        BASE_URL: ${{ secrets.BASE_URL }}
        TEST_EMAIL: ${{ secrets.TEST_EMAIL }}
        TEST_PASSWORD: ${{ secrets.TEST_PASSWORD }}
    
    - name: Upload Smoke Test Results - Chromium
      uses: actions/upload-artifact@v4
      if: always()
      with:
        name: smoke-test-results-chromium
        path: |
          playwright-report/
          test-results/
        retention-days: 7

  smoke-tests-firefox:
    # Similar configuration for Firefox
    # ...

  smoke-tests-webkit:
    # Similar configuration for WebKit
    # ...

  smoke-summary:
    needs: [smoke-tests-chromium, smoke-tests-firefox, smoke-tests-webkit]
    runs-on: ubuntu-latest
    if: always()
    
    steps:
    - name: Generate Smoke Tests Summary
      run: |
        echo "## 🔥 Smoke Tests Complete Summary" >> $GITHUB_STEP_SUMMARY
        echo "" >> $GITHUB_STEP_SUMMARY
        echo "**Scope:** ${{ github.event.inputs.test_scope || 'core' }}" >> $GITHUB_STEP_SUMMARY
        echo "**Browsers:** Chromium, Firefox, WebKit" >> $GITHUB_STEP_SUMMARY
        echo "**Execution Time:** $(date)" >> $GITHUB_STEP_SUMMARY
        echo "" >> $GITHUB_STEP_SUMMARY
        echo "### 📊 Test Scopes:" >> $GITHUB_STEP_SUMMARY
        echo "- **Core:** TC-001 a TC-011 (Login y autenticación)" >> $GITHUB_STEP_SUMMARY
        echo "- **Extended:** TC-001 a TC-014 (Core + funcionalidades básicas)" >> $GITHUB_STEP_SUMMARY
        echo "- **All:** TC-001 a TC-016 (Todos los smoke tests)" >> $GITHUB_STEP_SUMMARY
        echo "" >> $GITHUB_STEP_SUMMARY
        echo "### 🎯 Objetivo: Validar funcionalidad básica según Excel STD_KPIs" >> $GITHUB_STEP_SUMMARY
```
